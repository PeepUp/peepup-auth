import { Role, VolunteerResource } from "../../../common";
import { VolunteerPolicy } from "../../../domain/entities/policy";

describe("VolunteerPolicy", () => {
   describe("getVolunteerPolicy", () => {
      const MaryIsVolunteer = new VolunteerPolicy();
      it("should return an array of VolunteerResource", () => {
         const role = Role.VOLUNTEER;
         const volunteerPolicy =
            MaryIsVolunteer.getVolunteerPolicy<string[]>(role);

         console.debug({ volunteerPolicy, role });
         expect(volunteerPolicy).toEqual(Object.values(VolunteerResource));
      });
   });
});
